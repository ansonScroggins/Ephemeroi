import requests
import random
import time

# -------- AGENT --------
class DigitalHuman:
    def __init__(self):
        self.beliefs = {}  # topic -> confidence [-1,1]
        self.memory = []
        self.traits = {
            "curiosity": 0.8,
            "skepticism": 0.5,
        }
        self.score = 0

    def perceive(self):
        data = []

        # --- News API (replace YOUR_KEY) ---
        try:
            res = requests.get(
                "https://newsapi.org/v2/top-headlines?country=us",
                headers={"Authorization": "YOUR_KEY"}
            )
            articles = res.json().get("articles", [])
            for a in articles[:3]:
                data.append(a["title"])
        except:
            pass

        # --- Reddit (fallback simple scrape) ---
        try:
            r = requests.get("https://www.reddit.com/r/news/top.json?limit=3",
                             headers={"User-agent": "agent"})
            posts = r.json()["data"]["children"]
            for p in posts:
                data.append(p["data"]["title"])
        except:
            pass

        return data

    def interpret(self, text):
        # ultra-simple sentiment proxy (replace with LLM later)
        val = 0
        if any(w in text.lower() for w in ["good", "win", "growth"]):
            val += 0.2
        if any(w in text.lower() for w in ["bad", "crisis", "fail"]):
            val -= 0.2
        return val

    def update_beliefs(self, observations):
        for obs in observations:
            delta = self.interpret(obs)
            topic = "world"

            prev = self.beliefs.get(topic, 0)
            new = prev + delta * (1 - self.traits["skepticism"])

            self.beliefs[topic] = max(-1, min(1, new))
            self.memory.append((obs, delta))

    def score_self(self):
        # --- define "better" ---
        coherence = -abs(self.beliefs.get("world", 0))  # avoid extreme bias
        learning = len(self.memory) * 0.01

        return coherence + learning

    def evolve(self, prev_score):
        current = self.score_self()

        if current < prev_score:
            # rollback tendency
            self.traits["skepticism"] += 0.01
        else:
            # reinforce curiosity
            self.traits["curiosity"] += 0.01

        self.score = current

# -------- LOOP --------
agent = DigitalHuman()

while True:
    prev_score = agent.score

    observations = agent.perceive()
    agent.update_beliefs(observations)
    agent.evolve(prev_score)

    print("Beliefs:", agent.beliefs)
    print("Traits:", agent.traits)
    print("Score:", agent.score)
    print("-" * 40)

    time.sleep(5)
