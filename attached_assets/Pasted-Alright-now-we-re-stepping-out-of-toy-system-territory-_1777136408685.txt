🧬 Agent Store

# agents.py
import numpy as np
class AgentStore:
    def __init__(self):
        self.agents = {}
    def create(self, name):
        vec = np.random.randn(384)
        agent_id = str(len(self.agents))
        self.agents[agent_id] = {
            "name": name,
            "vector": vec,
            "history": [],
            "confidence": 0.5
        }
        return self.agents[agent_id]
    def update(self, agent_id, content_vec, feedback):
        agent = self.agents[agent_id]
        agent["vector"] += feedback * content_vec

🧠 Influence Function

# influence.py
import numpy as np
def interact(a_id, b_id, store):
    A = store.agents[a_id]
    B = store.agents[b_id]
    similarity = np.dot(A["vector"], B["vector"])
    influence_strength = (1 - A["confidence"]) * similarity
    # A moves toward B
    A["vector"] += influence_strength * (B["vector"] - A["vector"])
    # B resists or adapts slightly
    B["vector"] += 0.1 * influence_strength * (A["vector"] - B["vector"])
    return {
        "similarity": float(similarity),
        "influence": float(influence_strength)
    }
