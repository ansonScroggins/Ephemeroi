🌊 2) Ingestion Worker (GH Archive example)
Pull hourly GitHub events and push into Redis Stream.
# worker/ingest_gharchive.py
import aiohttp, asyncio, json, datetime
import redis.asyncio as redis

REDIS_URL = "redis://redis:6379"
STREAM = "ephemeroi:raw"

async def fetch_hour(dt):
    url = f"https://data.gharchive.org/{dt:%Y-%m-%d-%-H}.json.gz"
    async with aiohttp.ClientSession() as s:
        async with s.get(url) as r:
            if r.status != 200:
                return
            import gzip
            data = gzip.decompress(await r.read()).decode().splitlines()
            return [json.loads(x) for x in data]

async def main():
    r = redis.from_url(REDIS_URL)
    dt = datetime.datetime.utcnow() - datetime.timedelta(hours=1)

    events = await fetch_hour(dt)
    if not events:
        return

    for ev in events:
        await r.xadd(STREAM, {"data": json.dumps(ev)}, maxlen=100000)

asyncio.run(main())
Run this on a cron (every hour).

Consumes stream → emits telemetry events.
# worker/analyzer.py
import asyncio, json, redis.asyncio as redis
from math import tanh

REDIS_URL = "redis://redis:6379"
IN_STREAM = "ephemeroi:raw"
OUT_STREAM = "ephemeroi:signals"

def compute_metrics(ev):
    # crude example → replace with your real model
    is_issue = ev.get("type") == "IssuesEvent"
    comments = ev.get("payload", {}).get("issue", {}).get("comments", 0)

    consensus = tanh(comments / 10)
    unsat = 1 if is_issue else 0

    cage = consensus > 0.9 and unsat > 0

    return {
        "consensus": consensus,
        "unsat": unsat,
        "cage": cage,
        "pressure": unsat - consensus
    }

async def main():
    r = redis.from_url(REDIS_URL)
    last_id = "0-0"

    while True:
        resp = await r.xread({IN_STREAM: last_id}, block=5000, count=100)
        if not resp:
            continue

        for stream, messages in resp:
            for msg_id, fields in messages:
                last_id = msg_id
                ev = json.loads(fields[b"data"])

                m = compute_metrics(ev)

                signal = {
                    "event": "STRUCTURAL_REALIGNMENT",
                    "repo": ev.get("repo", {}).get("name"),
                    "metrics": m
                }

                await r.xadd(OUT_STREAM, {"data": json.dumps(signal)})
                
asyncio.run(main())

# worker/verification.py
def compute_trust(signal):
    m = signal["metrics"]
    trust = 1.0

    if m["cage"]:
        trust -= 0.3
    if m["pressure"] > 0.5:
        trust -= 0.2

    return max(trust, 0.0)

# bot/main.py
import os, asyncio, json
import redis.asyncio as redis
from telegram import Bot

TOKEN = os.environ["TELEGRAM_TOKEN"]
CHAT_ID = os.environ["CHAT_ID"]
REDIS_URL = os.environ["REDIS_URL"]

bot = Bot(token=TOKEN)

async def main():
    r = redis.from_url(REDIS_URL)
    last_id = "0-0"

    while True:
        resp = await r.xread({"ephemeroi:signals": last_id}, block=5000)
        if not resp:
            continue

        for _, messages in resp:
            for msg_id, fields in messages:
                last_id = msg_id
                signal = json.loads(fields[b"data"])

                m = signal["metrics"]

                if not (m["cage"] or m["pressure"] > 0.5):
                    continue

                text = f"""
🌌 EPHEMEROI SIGNAL

Repo: {signal['repo']}
Consensus: {m['consensus']:.2f}
Unsat: {m['unsat']}
Pressure: {m['pressure']:.2f}

⚠️ Cage Detected: {m['cage']}
"""

                await bot.send_message(chat_id=CHAT_ID, text=text)

asyncio.run(main())
