// src/lib/ephemeroi/crackLeverager.ts

export type IssueEvent = {
  id: number;
  title: string;
  body?: string;
  labels?: string[];
  state: "open" | "closed";
  created_at: number;
  closed_at?: number;
};

export type Crack = {
  signature: string;
  frequency: number;
  lastSeen: number;
  severity: number;
};

export type LeverageReport = {
  crack: Crack;
  classification: string;
  prediction: string;
  leverage: string[];
};

export class CrackLeverager {
  private cracks: Map<string, Crack> = new Map();

  ingest(issue: IssueEvent) {
    const signature = this.extractSignature(issue);
    const now = Date.now();

    const existing = this.cracks.get(signature);

    if (existing) {
      existing.frequency += 1;
      existing.lastSeen = now;
      existing.severity = this.computeSeverity(existing);
    } else {
      this.cracks.set(signature, {
        signature,
        frequency: 1,
        lastSeen: now,
        severity: 10
      });
    }
  }

  generateReports(): LeverageReport[] {
    return Array.from(this.cracks.values()).map(crack => {
      const classification = this.classify(crack.signature);
      const prediction = this.predict(crack);
      const leverage = this.generateLeverage(crack, classification);

      return { crack, classification, prediction, leverage };
    });
  }

  // --- Core Intelligence ---

  private extractSignature(issue: IssueEvent): string {
    const text = `${issue.title} ${issue.body ?? ""}`.toLowerCase();

    if (text.includes("preproc")) return "preprocessing";
    if (text.includes("cli") || text.includes("flag")) return "cli-interface";
    if (text.includes("crash") || text.includes("segfault")) return "runtime-crash";
    if (text.includes("incorrect") || text.includes("wrong")) return "logic-error";

    return "unknown";
  }

  private classify(signature: string): string {
    switch (signature) {
      case "preprocessing":
        return "Usability Instability in Preprocessing Pipeline";
      case "cli-interface":
        return "Configuration Surface Fragility";
      case "runtime-crash":
        return "Execution Integrity Failure";
      case "logic-error":
        return "Correctness Deviation";
      default:
        return "Unclassified System Weakness";
    }
  }

  private predict(crack: Crack): string {
    if (crack.frequency > 5) {
      return "High likelihood of recurring user-facing failures";
    }
    if (crack.severity > 50) {
      return "Potential systemic design flaw";
    }
    return "Localized issue, monitor for spread";
  }

  private generateLeverage(crack: Crack, classification: string): string[] {
    if (classification.includes("Preprocessing")) {
      return [
        "Build wrapper enforcing safe defaults",
        "Add pre-run validation checks",
        "Document actual vs expected behavior"
      ];
    }

    if (classification.includes("Configuration")) {
      return [
        "Create config schema validator",
        "Introduce strict flag parsing",
        "Surface misconfigurations early"
      ];
    }

    if (classification.includes("Execution")) {
      return [
        "Add runtime guards",
        "Implement crash telemetry",
        "Isolate failing components"
      ];
    }

    return [
      "Collect more data",
      "Cluster similar issues",
      "Refine classification model"
    ];
  }

  private computeSeverity(crack: Crack): number {
    return Math.min(100, crack.frequency * 10);
  }
}
